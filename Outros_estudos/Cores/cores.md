# Como usar:
\033[ "info" m "texto" \033[m

Entre o m e o colchete coloca as informaçoes

"\033[**0;30;41**mTeste\033[m"


# Estilos:
0 = None (sem estilo) <br>
1 = **Bold (Negrito)** <br>
2 = Fraco <br>
3 = _Itálico_ <br>
4 = Underline (Sublinhado) <br>
7 = Negative (Inverter as configurações Fundo vai pra letra e letra fundo) <br>

# Cores de texto:
30 = Branco <br>
31 = Vermelho * <br>
32 = Verde * <br>
33 = Amarelo <br>
34 = Azul * <br>
35 = Magenta <br>
36 = Ciano <br> 
37 = Cinza <br>

# Cores de fundo:
40 = Branco <br>
41 = Vermelho * <br>
42 = Verde * <br>
43 = Amarelo <br>
44 = Azul * <br>
45 = Magenta <br>
46 = Ciano <br>
47 = Cinza <br>
